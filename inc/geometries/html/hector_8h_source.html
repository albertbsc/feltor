130  const thrust::host_vector<double>& zetaU, //2d Zeta component
131  const thrust::host_vector<double>& etaU, //2d Eta component
132  const thrust::host_vector<double>& u_vec, //1d u values

134  const thrust::host_vector<double>& eta_init, //1d intial values

141  unsigned N = 1;
142  double eps = 1e10, eps_old=2e10;
143  std::vector<thrust::host_vector<double> > begin(2);
144  begin[0] = zeta_init, begin[1] = eta_init;
145  //now we have the starting values
146  std::vector<thrust::host_vector<double> > end(begin), temp(begin);
147  unsigned sizeU = u_vec.size(), sizeV = zeta_init.size();
148  unsigned size2d = sizeU*sizeV;
149  zeta.resize(size2d), eta.resize(size2d);
150  double u0=0, u1 = u_vec[0];
151  thrust::host_vector<double> zeta_old(zeta), zeta_diff( zeta), eta_old(eta), eta_diff(eta);
152  while( (eps < eps_old || eps > 1e-6) && eps > 1e-7)
153  {
154  zeta_old = zeta, eta_old = eta; eps_old = eps;
155  temp = begin;
157  for( unsigned i=0; i<sizeU; i++)
158  {
159  u0 = i==0?0:u_vec[i-1], u1 = u_vec[i];
160  dg::stepperRK17( inter, temp, end, u0, u1, N);
161  for( unsigned j=0; j<sizeV; j++)
162  {
163  unsigned idx = j*sizeU+i;
164  zeta[idx] = end[0][j], eta[idx] = end[1][j];
165  }
166  temp = end;
167  }
168  dg::blas1::axpby( 1., zeta, -1., zeta_old, zeta_diff);

170  dg::blas1::pointwiseDot( zeta_diff, zeta_diff, zeta_diff);
171  dg::blas1::pointwiseDot( 1., eta_diff, eta_diff, 1., zeta_diff);
172  eps = sqrt( dg::blas1::dot( zeta_diff, zeta_diff)/sizeU/sizeV);

555  dg::blas1::transfer( chi_ZE, chiZE);
556  dg::blas1::transfer( chi_EE, chiEE);

566  dg::blas1::pointwiseDot( 1., u_eta, temp_eta, 1., temp_scalar);

568  dg::blas1::pointwiseDivide( zetaU, temp_scalar, zetaU);
569  dg::blas1::pointwiseDivide( etaU, temp_scalar, etaU);
570  //now compute etaV and its inverse
571  container etaVinv(u_zeta), etaV(etaVinv);
572  dg::blas1::pointwiseDot( etaVinv, chiZZ, etaVinv);
573  dg::geo::multiplyPerpVolume( etaVinv, g2d_);

580  dg::blas1::axpby( +1., temp_zeta, 0.,v_eta);

590  dg::blas1::scal( zetaU, 1./c0_);
591  dg::blas1::scal( etaU, 1./c0_);
592  dg::blas1::scal( u_zeta, c0_);
593  dg::blas1::scal( v_zeta, c0_);

600  dg::blas1::transfer( etaU, etaU_);
601  dg::blas1::transfer( zetaU, zetaU_);
