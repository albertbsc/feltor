400  if( turns ==0) turns++;

413  if( sizeXY != 1){

416  }

419  temp.swap( tempP);

432  }

439  if(idx>=0)
440  result[i0] = plus2d[idx];
441  else
442  result[i0] = minus2d[abs(idx)];
443  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.data().begin() + i0*g2d.size());
444  }
445  }

460  }
461  }
