603 int integrateRK(RHS& rhs, const Vector& begin, Vector& end, double T_max, double eps_abs )
604 {
605  RK_classic<s, Vector > rk( begin);
606  Vector old_end(begin), temp(begin);
607  end = begin;
608  if( T_max == 0) return 0;
609  double dt = T_max/10;
610  int NT = 10;
611  double error = 1e10;
612 
613  while( error > eps_abs && NT < pow( 2, 18) )
614  {
615  dt /= 2.;
616  NT *= 2;
617  end = begin;
618 
619  int i=0;
620  while (i<NT)
621  {
622  rk( rhs, end, temp, dt);
623  if( !rhs.monitor( temp ) ) //sanity check
624  {

628  break;
629  }
630  end.swap( temp); //end is one step further
631  i++;
632  }

634  old_end = end;
635 #ifdef DG_DEBUG

640 #endif //MPI

642 #endif //DG_DEBUG
643  }

645  if( std::isnan( error) )

648  return -1;
