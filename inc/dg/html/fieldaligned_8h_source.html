445 container FieldAligned<G,M,container>::evaluate( BinaryOp binary, UnaryOp unary, unsigned p0, unsigned rounds) const
446 {
447  //idea: simply apply I+/I- enough times on the init2d vector to get the result in each plane
448  //unary function is always such that the p0 plane is at x=0
449  assert( g_.Nz() > 1);
450  assert( p0 < g_.Nz());
451  const typename G::perpendicular_grid g2d = g_.perp_grid();
452  container init2d = dg::pullback( binary, g2d), temp(init2d), tempP(init2d), tempM(init2d);
453  container vec3d = dg::evaluate( dg::zero, g_);
454  std::vector<container> plus2d( g_.Nz(), (container)dg::evaluate(dg::zero, g2d) ), minus2d( plus2d), result( plus2d);
455  unsigned turns = rounds;
456  if( turns ==0) turns++;
457  //first apply Interpolation many times, scale and store results
458  for( unsigned r=0; r<turns; r++)
459  for( unsigned i0=0; i0<g_.Nz(); i0++)
460  {
461  dg::blas1::copy( init2d, tempP);
462  dg::blas1::copy( init2d, tempM);
463  unsigned rep = i0 + r*g_.Nz();
464  for(unsigned k=0; k<rep; k++)
465  {
466  dg::blas2::symv( plus, tempP, temp);
467  temp.swap( tempP);
468  dg::blas2::symv( minus, tempM, temp);
469  temp.swap( tempM);
470  }
471  dg::blas1::scal( tempP, unary( (double)rep*g_.hz() ) );
472  dg::blas1::scal( tempM, unary( -(double)rep*g_.hz() ) );
473  dg::blas1::axpby( 1., tempP, 1., plus2d[i0]);
474  dg::blas1::axpby( 1., tempM, 1., minus2d[i0]);
475  }
476  //now we have the plus and the minus filaments
477  if( rounds == 0) //there is a limiter
478  {
479  for( unsigned i0=0; i0<g_.Nz(); i0++)
480  {
481  int idx = (int)i0 - (int)p0;
482  if(idx>=0)
483  result[i0] = plus2d[idx];
484  else
485  result[i0] = minus2d[abs(idx)];
486  thrust::copy( result[i0].begin(), result[i0].end(), vec3d.begin() + i0*g2d.size());
487  }
488  }
489  else //sum up plus2d and minus2d
490  {
491  for( unsigned i0=0; i0<g_.Nz(); i0++)
492  {
493  unsigned revi0 = (g_.Nz() - i0)%g_.Nz(); //reverted index
494  dg::blas1::axpby( 1., plus2d[i0], 0., result[i0]);
495  dg::blas1::axpby( 1., minus2d[revi0], 1., result[i0]);
496  }

498  for(unsigned i0=0; i0<g_.Nz(); i0++)
499  {

502  }
